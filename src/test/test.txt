<!-- 基本的なタグ -->
<p>これは段落です。</p>

<!-- 複数のタグとテキスト -->
<div>
  <h1>見出し</h1>
  <p>これは<b>太字</b>のテキストです。</p>
</div>

<!-- ネストされたタグ -->
<div>
  <ul>
    <li>リストアイテム1</li>
    <li>リストアイテム2</li>
  </ul>
</div>

<!-- 属性を含むタグ -->
<div class="container">
  <p id="paragraph">ここは<span style="color: red;">赤いテキスト</span>です。</p>
</div>

<!-- 自己閉じタグ -->
<p>この文には<br />改行が含まれています。<img src="image.jpg" /></p>

<!-- 空のタグ -->
<p>ここには何も<b></b>含まれていません。</p>

<!-- テキストにタグが含まれていないケース -->
<p>ただのテキスト、タグは含まれていません。</p>
